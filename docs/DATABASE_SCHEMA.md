# Database Schema Documentation\n\n## Overview\n\nSimCoder Pro uses PostgreSQL 15 as the primary database with a comprehensive schema to support all application features including coding simulation, performance tracking, training, and auditing.\n\n## Database Tables\n\n### Users Table\n```sql\nCREATE TABLE users (\n  id SERIAL PRIMARY KEY,\n  email VARCHAR(255) UNIQUE NOT NULL,\n  password_hash VARCHAR(255) NOT NULL,\n  first_name VARCHAR(100) NOT NULL,\n  last_name VARCHAR(100) NOT NULL,\n  avatar_url TEXT,\n  career_level INT DEFAULT 1,\n  current_specialty VARCHAR(50),\n  total_charts_coded INT DEFAULT 0,\n  total_accuracy DECIMAL(5,2) DEFAULT 0,\n  total_denials DECIMAL(5,2) DEFAULT 0,\n  account_created_at TIMESTAMP DEFAULT NOW(),\n  last_login_at TIMESTAMP,\n  status VARCHAR(20) DEFAULT 'active',\n  theme VARCHAR(20) DEFAULT 'light',\n  notification_preferences JSON,\n  created_at TIMESTAMP DEFAULT NOW(),\n  updated_at TIMESTAMP DEFAULT NOW()\n);\n```\n\n### Charts Table\n```sql\nCREATE TABLE charts (\n  id SERIAL PRIMARY KEY,\n  user_id INT NOT NULL REFERENCES users(id),\n  chart_type VARCHAR(50) NOT NULL, -- office_visit, procedure, consultation, etc.\n  specialty VARCHAR(50) NOT NULL,\n  medical_documentation TEXT NOT NULL,\n  documentation_date DATE,\n  patient_name VARCHAR(100),\n  patient_dob DATE,\n  provider_name VARCHAR(100),\n  provider_specialty VARCHAR(50),\n  facility_name VARCHAR(100),\n  difficulty_level VARCHAR(20), -- beginner, intermediate, advanced, expert, master\n  chart_status VARCHAR(20) DEFAULT 'assigned', -- assigned, in_progress, submitted, audited\n  user_submitted_codes JSON,\n  correct_codes JSON,\n  accuracy_score DECIMAL(5,2),\n  time_spent_seconds INT,\n  submitted_at TIMESTAMP,\n  audit_status VARCHAR(20),\n  audit_score DECIMAL(5,2),\n  audit_feedback TEXT,\n  created_at TIMESTAMP DEFAULT NOW(),\n  updated_at TIMESTAMP DEFAULT NOW()\n);\n```\n\n### ICD-10-CM Codes Table\n```sql\nCREATE TABLE icd10_codes (\n  id SERIAL PRIMARY KEY,\n  code VARCHAR(10) UNIQUE NOT NULL,\n  description VARCHAR(500) NOT NULL,\n  long_description TEXT,\n  chapter INT,\n  section VARCHAR(10),\n  category VARCHAR(10),\n  subcategory VARCHAR(10),\n  excludes1 TEXT,\n  excludes2 TEXT,\n  includes TEXT,\n  code_first TEXT,\n  use_additional_code TEXT,\n  requires_7th_character BOOLEAN DEFAULT FALSE,\n  seventh_character_options TEXT,\n  guidelines_note TEXT,\n  clinical_examples TEXT,\n  common_scenarios TEXT,\n  effective_date DATE,\n  deleted_date DATE,\n  created_at TIMESTAMP DEFAULT NOW(),\n  updated_at TIMESTAMP DEFAULT NOW()\n);\n\nCREATE INDEX idx_icd10_code ON icd10_codes(code);\nCREATE INDEX idx_icd10_chapter ON icd10_codes(chapter);\nCREATE INDEX idx_icd10_description ON icd10_codes USING GIN(to_tsvector('english', description));\n```\n\n### CPT Codes Table\n```sql\nCREATE TABLE cpt_codes (\n  id SERIAL PRIMARY KEY,\n  code VARCHAR(10) UNIQUE NOT NULL,\n  descriptor VARCHAR(500) NOT NULL,\n  long_descriptor TEXT,\n  category INT,\n  section VARCHAR(50),\n  subsection VARCHAR(100),\n  type VARCHAR(50),\n  base_unit DECIMAL(10,2),\n  work_unit DECIMAL(10,2),\n  pe_unit DECIMAL(10,2),\n  malpractice_unit DECIMAL(10,2),\n  transitioned_code VARCHAR(10),\n  active BOOLEAN DEFAULT TRUE,\n  status_indicator VARCHAR(20),\n  bilateral_indicator VARCHAR(20),\n  restrictions TEXT,\n  bundling_rules TEXT,\n  common_modifiers TEXT,\n  typical_uses TEXT,\n  guidelines_note TEXT,\n  clinical_examples TEXT,\n  common_scenarios TEXT,\n  effective_date DATE,\n  deleted_date DATE,\n  created_at TIMESTAMP DEFAULT NOW(),\n  updated_at TIMESTAMP DEFAULT NOW()\n);\n\nCREATE INDEX idx_cpt_code ON cpt_codes(code);\nCREATE INDEX idx_cpt_category ON cpt_codes(category);\nCREATE INDEX idx_cpt_descriptor ON cpt_codes USING GIN(to_tsvector('english', descriptor));\n```\n\n### HCPCS Codes Table\n```sql\nCREATE TABLE hcpcs_codes (\n  id SERIAL PRIMARY KEY,\n  code VARCHAR(10) UNIQUE NOT NULL,\n  description VARCHAR(500) NOT NULL,\n  long_description TEXT,\n  category VARCHAR(50),\n  subcategory VARCHAR(100),\n  type VARCHAR(50),\n  unit_description VARCHAR(100),\n  coverage_notes TEXT,\n  restrictions TEXT,\n  modifiers_allowed TEXT,\n  bundling_rules TEXT,\n  common_uses TEXT,\n  guidelines_note TEXT,\n  clinical_examples TEXT,\n  common_scenarios TEXT,\n  effective_date DATE,\n  deleted_date DATE,\n  created_at TIMESTAMP DEFAULT NOW(),\n  updated_at TIMESTAMP DEFAULT NOW()\n);\n\nCREATE INDEX idx_hcpcs_code ON hcpcs_codes(code);\nCREATE INDEX idx_hcpcs_category ON hcpcs_codes(category);\n```\n\n### Modifiers Table\n```sql\nCREATE TABLE code_modifiers (\n  id SERIAL PRIMARY KEY,\n  modifier_code VARCHAR(3) UNIQUE NOT NULL,\n  short_description VARCHAR(200) NOT NULL,\n  long_description TEXT,\n  applicable_to VARCHAR(50),\n  usage_rules TEXT,\n  when_to_use TEXT,\n  when_not_to_use TEXT,\n  examples TEXT,\n  common_mistakes TEXT,\n  related_modifiers TEXT,\n  created_at TIMESTAMP DEFAULT NOW(),\n  updated_at TIMESTAMP DEFAULT NOW()\n);\n\nCREATE INDEX idx_modifier_code ON code_modifiers(modifier_code);\n```\n\n### Performance Metrics Table\n```sql\nCREATE TABLE performance_metrics (\n  id SERIAL PRIMARY KEY,\n  user_id INT NOT NULL REFERENCES users(id),\n  metric_date DATE NOT NULL,\n  charts_coded INT DEFAULT 0,\n  total_accuracy DECIMAL(5,2),\n  audit_score DECIMAL(5,2),\n  denial_rate DECIMAL(5,2),\n  average_coding_time_seconds INT,\n  charts_completed INT DEFAULT 0,\n  charts_remaining INT DEFAULT 0,\n  performance_rating VARCHAR(20),\n  created_at TIMESTAMP DEFAULT NOW(),\n  updated_at TIMESTAMP DEFAULT NOW(),\n  UNIQUE(user_id, metric_date)\n);\n\nCREATE INDEX idx_performance_user ON performance_metrics(user_id);\nCREATE INDEX idx_performance_date ON performance_metrics(metric_date);\n```\n\n### Audits Table\n```sql\nCREATE TABLE audits (\n  id SERIAL PRIMARY KEY,\n  user_id INT NOT NULL REFERENCES users(id),\n  chart_id INT NOT NULL REFERENCES charts(id),\n  audit_date TIMESTAMP DEFAULT NOW(),\n  audit_result VARCHAR(20), -- pass, fail, partial\n  error_count INT DEFAULT 0,\n  error_details JSON,\n  error_types TEXT[], -- wrong_diagnosis, missing_code, modifier_error, sequencing_error, etc.\n  suggested_corrections JSON,\n  auditor_notes TEXT,\n  audit_score DECIMAL(5,2),\n  learning_resources_recommended TEXT[],\n  created_at TIMESTAMP DEFAULT NOW(),\n  updated_at TIMESTAMP DEFAULT NOW()\n);\n\nCREATE INDEX idx_audit_user ON audits(user_id);\nCREATE INDEX idx_audit_chart ON audits(chart_id);\nCREATE INDEX idx_audit_date ON audits(audit_date);\n```\n\n### Training Content Table\n```sql\nCREATE TABLE training_content (\n  id SERIAL PRIMARY KEY,\n  module_id VARCHAR(100) UNIQUE NOT NULL,\n  title VARCHAR(255) NOT NULL,\n  description TEXT,\n  content_type VARCHAR(50), -- video, lesson, quiz, case_study, interactive\n  difficulty_level VARCHAR(20), -- beginner, intermediate, advanced, expert\n  estimated_duration_minutes INT,\n  learning_objectives TEXT[],\n  content BYTEA,\n  prerequisites TEXT[],\n  related_codes TEXT[],\n  order_position INT,\n  is_active BOOLEAN DEFAULT TRUE,\n  created_at TIMESTAMP DEFAULT NOW(),\n  updated_at TIMESTAMP DEFAULT NOW()\n);\n\nCREATE INDEX idx_training_module ON training_content(module_id);\n```\n\n### User Progress Table\n```sql\nCREATE TABLE user_progress (\n  id SERIAL PRIMARY KEY,\n  user_id INT NOT NULL REFERENCES users(id),\n  training_id INT NOT NULL REFERENCES training_content(id),\n  progress_status VARCHAR(20) DEFAULT 'not_started', -- not_started, in_progress, completed\n  started_at TIMESTAMP,\n  completed_at TIMESTAMP,\n  score DECIMAL(5,2),\n  time_spent_seconds INT,\n  notes TEXT,\n  bookmarked BOOLEAN DEFAULT FALSE,\n  created_at TIMESTAMP DEFAULT NOW(),\n  updated_at TIMESTAMP DEFAULT NOW(),\n  UNIQUE(user_id, training_id)\n);\n\nCREATE INDEX idx_progress_user ON user_progress(user_id);\nCREATE INDEX idx_progress_status ON user_progress(progress_status);\n```\n\n### NCCI Edits Table\n```sql\nCREATE TABLE ncci_edits (\n  id SERIAL PRIMARY KEY,\n  primary_code VARCHAR(10) NOT NULL,\n  secondary_code VARCHAR(10) NOT NULL,\n  edit_type VARCHAR(50),\n  ncci_indicator VARCHAR(20),\n  edit_description TEXT,\n  rationale TEXT,\n  modifier_to_bypass VARCHAR(5),\n  effective_date DATE,\n  created_at TIMESTAMP DEFAULT NOW(),\n  FOREIGN KEY(primary_code) REFERENCES cpt_codes(code),\n  FOREIGN KEY(secondary_code) REFERENCES cpt_codes(code)\n);\n\nCREATE INDEX idx_ncci_primary ON ncci_edits(primary_code);\nCREATE INDEX idx_ncci_secondary ON ncci_edits(secondary_code);\n```\n\n### Coverage Requirements Table\n```sql\nCREATE TABLE coverage_requirements (\n  id SERIAL PRIMARY KEY,\n  code_type VARCHAR(20), -- ICD-10, CPT, HCPCS\n  code VARCHAR(10) NOT NULL,\n  payer_type VARCHAR(50), -- Medicare, Medicaid, Commercial, etc.\n  coverage_status VARCHAR(50), -- Covered, Not Covered, Conditional\n  requirements TEXT,\n  documentation_needed TEXT,\n  denial_reasons TEXT,\n  prior_auth_required BOOLEAN,\n  quantity_limits INT,\n  age_restrictions TEXT,\n  gender_restrictions VARCHAR(20),\n  notes TEXT,\n  effective_date DATE,\n  created_at TIMESTAMP DEFAULT NOW(),\n  updated_at TIMESTAMP DEFAULT NOW()\n);\n\nCREATE INDEX idx_coverage_code ON coverage_requirements(code);\nCREATE INDEX idx_coverage_payer ON coverage_requirements(payer_type);\n```\n\n### Career Progression Table\n```sql\nCREATE TABLE career_progression (\n  id SERIAL PRIMARY KEY,\n  user_id INT NOT NULL REFERENCES users(id),\n  level INT NOT NULL, -- 1-10\n  level_name VARCHAR(100),\n  requirements_met JSON,\n  accuracy_percentage DECIMAL(5,2),\n  charts_required INT,\n  audit_score_required DECIMAL(5,2),\n  achieved_date TIMESTAMP,\n  created_at TIMESTAMP DEFAULT NOW(),\n  UNIQUE(user_id, level)\n);\n\nCREATE INDEX idx_career_user ON career_progression(user_id);\n```\n\n### Achievements Table\n```sql\nCREATE TABLE achievements (\n  id SERIAL PRIMARY KEY,\n  user_id INT NOT NULL REFERENCES users(id),\n  achievement_type VARCHAR(100),\n  achievement_name VARCHAR(255),\n  description TEXT,\n  badge_icon TEXT,\n  earned_at TIMESTAMP DEFAULT NOW(),\n  created_at TIMESTAMP DEFAULT NOW()\n);\n\nCREATE INDEX idx_achievement_user ON achievements(user_id);\n```\n\n## Database Relationships\n\n```\nusers (1) --> (many) charts\nusers (1) --> (many) performance_metrics\nusers (1) --> (many) audits\nusers (1) --> (many) user_progress\nusers (1) --> (many) career_progression\nusers (1) --> (many) achievements\n\ncharts (many) --> (1) audits\ntraining_content (many) --> (1) user_progress\n```\n\n## Indexing Strategy\n\n- Primary lookups: Code tables indexed on code value\n- Searches: Full-text indexes on description fields\n- User queries: Foreign key indexes on user_id\n- Date-based queries: Indexes on date columns\n- Performance queries: Composite indexes on frequently used combinations\n\n## Data Partitioning (Future)\n\n- Charts table: Partition by year\n- Performance metrics: Partition by month\n- Audits: Partition by year\n\n## Backup Strategy\n\n- Daily full backups\n- Hourly incremental backups\n- Off-site backup storage\n- Recovery time objective (RTO): 4 hours\n- Recovery point objective (RPO): 1 hour\n\n## Growth Projections\n\n- Expected 10,000+ users in year 1\n- Average 50+ charts per user annually\n- 5+ years of historical data retention\n- Estimated database size: 50GB in year 1, 200GB+ by year 3\n"
